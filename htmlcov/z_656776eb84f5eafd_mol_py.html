39def parse_mol_v2000(data): 

40 line = data[3] 

41 

42 atoms_count = int(line[0:3]) 

43 bonds_count = int(line[3:6]) 

44 if not atoms_count: 

45 raise EmptyMolecule 

46 

47 log = [] 

48 title = data[1].strip() or None 

49 atoms = [] 

50 bonds = [] 

51 stereo = [] 

52 hydrogens = {} 

53 dat = {} 

54 

55 for line in data[4: 4 + atoms_count]: 

56 try: 

57 charge = _charge_map[line[36:39]] 

58 except KeyError: 

59 raise InvalidCharge 

60 element = line[31:34].strip() 

61 isotope = line[34:36] 

62 

63 if element in 'AL': 

64 raise ValueError('queries not supported') 

65 elif element == 'D': 

66 element = 'H' 

67 if isotope != ' 0': 

68 raise ValueError('isotope on deuterium atom') 

69 isotope = 2 

70 elif isotope != ' 0': 

71 try: 

72 isotope = common_isotopes[element] + int(isotope) 

73 except KeyError: 

74 raise ValueError('invalid element symbol') 

75 else: 

76 isotope = None 

77 

78 mapping = line[60:63] 

79 atoms.append({'element': element, 'charge': charge, 'isotope': isotope, 'is_radical': False, 

80 'mapping': int(mapping) if mapping else 0, 'x': float(line[0:10]), 'y': float(line[10:20]), 

81 'z': float(line[20:30])}) 

82 

83 for line in data[4 + atoms_count: 4 + atoms_count + bonds_count]: 

84 a1, a2 = int(line[0:3]) - 1, int(line[3:6]) - 1 

85 s = line[9:12] 

86 if s == ' 1': 

87 stereo.append((a1, a2, 1)) 

88 elif s == ' 6': 

89 stereo.append((a1, a2, -1)) 

90 elif s != ' 0': 

91 log.append(f'unsupported or invalid stereo: {line}') 

92 b = int(line[6:9]) 

93 if b == 9: # added ad-hoc for bond type 9 

94 b = 8 

95 log.append(f'coordinate bond replaced with special: {line}') 

96 bonds.append((a1, a2, b)) 

97 

98 for line in data[4 + atoms_count + bonds_count:]: 

99 if line.startswith('M END'): 

100 break 

101 elif line.startswith('M ALS'): 

102 raise ValueError('list of atoms not supported') 

103 elif line.startswith(('M ISO', 'M RAD', 'M CHG')): 

104 _type = _ctf_data[line[3]] 

105 for i in range(int(line[6:9])): 

106 i8 = i * 8 

107 atom = int(line[10 + i8:13 + i8]) 

108 if not atom or atom > len(atoms): 

109 raise InvalidV2000('invalid atoms number') 

110 atom = atoms[atom - 1] 

111 atom[_type] = int(line[14 + i8:17 + i8]) 

112 

113 elif line.startswith('M STY'): 

114 for i in range(int(line[6:9])): 

115 i8 = i * 8 

116 if (st := line[14 + i8:17 + i8]) == 'DAT': 

117 dat[int(line[10 + i8:13 + i8])] = {} 

118 elif st == 'SUP': 

119 dat[int(line[10 + i8:13 + i8])] = {'type': 'MDL_SUP'} 

120 elif line.startswith('M SAL'): 

121 i = int(line[7:10]) 

122 if i in dat: 

123 dat[i]['atoms'] = tuple(int(line[14 + 4 * i:17 + 4 * i]) - 1 for i in range(int(line[10:13]))) 

124 elif line.startswith('M SDT'): 

125 i = int(line[7:10]) 

126 if i in dat: 

127 dat[i]['type'] = line.split()[-1].lower() 

128 elif line.startswith('M SED'): 

129 i = int(line[7:10]) 

130 if i in dat: 

131 dat[i]['value'] = line[10:].strip().replace('/', '').lower() 

132 elif line.startswith('M SMT'): 

133 i = int(line[7:10]) 

134 if i in dat: 

135 dat[i]['value'] = line[10:].strip() 

136 elif not line.startswith('M SDD'): 

137 log.append(f'ignored line: {line}') 

138 

139 for a in atoms: 

140 if a['is_radical']: # int to bool 

141 a['is_radical'] = True 

142 for x in dat.values(): 

143 try: 

144 _type = x['type'] 

145 if _type == 'mrv_implicit_h': 

146 _atoms = x['atoms'] 

147 value = x['value'] 

148 if len(_atoms) != 1 or _atoms[0] == -1 or not value: 

149 raise InvalidV2000(f'MRV_IMPLICIT_H spec invalid {x}') 

150 hydrogens[_atoms[0]] = int(value[6:]) 

151 else: 

152 log.append(f'ignored data: {x}') 

153 except KeyError: 

154 raise InvalidV2000(f'Invalid SGROUP {x}') 

155 

156 return {'title': title, 'atoms': atoms, 'bonds': bonds, 'stereo': stereo, 'hydrogens': hydrogens, 

157 'meta': None, 'log': log}
